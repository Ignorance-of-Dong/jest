<script>
import { COMPANY_LICENSE_UPLOAD } from "@/config/api";
import { licenseResourceAmount } from "~/config/apiControl";
import LicenseRecord from "~/components/License/LicenseRecord.vue";
import LicenseExpire from "~/components/License/LicenseExpire.vue";
import common from "~/mixins/common";
export default {
  data() {
    return {
      companyLicenseUpload: COMPANY_LICENSE_UPLOAD,
      licenseStatus: "",
      data: {},
      loading: true,
      tabActive: "detail",
      isReady: false
    };
  },
  mixins: [common],
  mounted() {
    this.getLicense();
  },
  methods: {
    detailGoBack() {
      this.isSeeLicenseDetail = false;
    },
    // 获取license信息
    async getLicense() {
      try {
        // let data = {
        //   tenantId: this.company.id
        // };
        this.loading = true;
        await this.$store.dispatch("license/getLicenseStatus");
        let res = await this.$store.dispatch("license/getLicenseDetail");
        this.isReady = true;
        let speace = await licenseResourceAmount({
          tenantId: this.company.id
        });
        if (res) {
          this.data = {
            ...res,
            ...speace
          };
          this.licenseStatus = "success";
          this.loading = false;
        }
      } catch (error) {
        this.loading = false;
        if (error && error.data && error.data.code == 10500032) {
          this.licenseStatus = "failed";
        }
      }
    },
    handleTabs() {},
    // 上传成功
    handleUploadSuccess(res) {
      if (res.code === 0) {
        // let data = res.data;
        this.getLicense();
      } else {
        this.$message.error(res.msg);
      }
    },
    handleUploadError() {
      // ..
    },
    beforeUploadUpload(file) {
      const isLt10KB = file.size / 1024 < 10;
      if (!isLt10KB) {
        this.$message.error(
          this.$t("common.message.error.imageSize", { size: "5MB" })
        );
        return false;
      }
    },
    formatData(num, unit = false) {
      if (num == -1) {
        return this.$t("common.license.unlimited");
      } else {
        return unit ? this.$t("common.license.count", { count: num }) : num;
      }
    }
  },
  computed: {
    userInfo() {
      return this.$store.state.account.userInfo || { username: "" };
    },
    company() {
      return this.$store.state.account.company || { name: "" };
    },
    status() {
      return (
        this.$store.state.license.licenseStatusDeatil &&
        this.$store.state.license.licenseStatusDeatil.status
      );
    },
    accountData() {
      return [
        {
          label: this.$t("account.fields.account"),
          value: this.userInfo.username
        },
        {
          label: this.$t("account.fields.company"),
          value: this.company.name
        }
      ];
    },
    transforColon() {
      return this.$i18n.locale == "en" ? ":" : "：";
    },
    licenseData() {
      return [
        {
          label: this.$t("common.license.versionInfo"),
          value: `${this.data["version"]} `
        },
        {
          label: this.$t("common.license.title") + this.transforColon,
          value: this.$t(
            `common.license.licenseTypes.${this.data.subscriptionType}`
          )
        },
        // {
        //   label: this.$t("common.license.userCount"),
        //   value: `${this.data["userCount"]} / ${this.formatData(
        //     this.data["maxUsers"]
        //   )} `
        // },
        {
          label: this.$t("common.license.licenseExpiryDate"),
          value: `${
            this.data["licenseExpiryDate"] == -1
              ? this.$t("common.license.unlimited")
              : this.formatTime(this.data["expireTime"], "yyyy-MM-dd")
          }`
        },
        {
          label: this.$t("common.license.eventOnlineCount"),
          value: this.formatData(this.data["maxEvents"])
        },
        {
          label: this.$t("common.license.eventOnlineMaxUsers"),
          value: this.formatData(this.data["eventMaxUsers"])
        },
        {
          label:
            this.$t("common.subRecords.recordStartTime") + this.transforColon,
          value: this.formatTime(this.data["startTime"], "yyyy-MM-dd hh:mm")
        },
        {
          label:
            this.$t("common.subRecords.recordEndtTime") + this.transforColon,
          value: this.formatTime(this.data["expireTime"], "yyyy-MM-dd hh:mm")
        },
        // {
        //   label: this.$t("common.license.maxModels"),
        //   value: `${this.formatData(this.data["maxModels"], true)}`
        // },
        // {
        //   label: this.$t("common.license.modelSize"),
        //   value: this.formatData(this.data["modelSize"]) + "MB"
        // },
        {
          label: this.$t("common.license.spaceUsage"),
          value: `${this.getFileSize(this.data["used"]) +
            " / " +
            this.getFileSize(this.data["maxSpaceLimit"] * 1024 * 1024)}`
        }
      ];
    },
    accountToken() {
      return this.$store.state.account.token;
    }
  },
  components: {
    LicenseRecord: LicenseRecord,
    LicenseExpire: LicenseExpire
  }
};
